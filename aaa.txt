  <script>
    // Daily streak system
    let streak = localStorage.getItem("streak") || 0;
    document.getElementById("streak").innerText = streak;

    function increaseStreak() {
      streak++;
      localStorage.setItem("streak", streak);
      document.getElementById("streak").innerText = streak;
      document.getElementById("playBtn").disabled = false; // unlock games
      alert("Great job! Your streak is now " + streak + " days 🔥");
    }

    // Avatar growth
    let avatarStage = 0;
    const avatars = ["🙂","😎","🧙‍♂️","👑"];
    function upgradeAvatar() {
      avatarStage = (avatarStage + 1) % avatars.length;
      document.getElementById("avatar").innerText = avatars[avatarStage];
    }

    // Quiz system
    function checkAnswer(ans) {
      let result = document.getElementById("quizResult");
      if (ans === 8) {
        result.innerText = "✅ Correct! Well done.";
      } else {
        result.innerText = "❌ Try again!";
      }
    }

    // Play unlock
    function playGame() {
      alert("🎮 Game loading... Have fun!");
    }
